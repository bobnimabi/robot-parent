/**
 * 字符串占位符格式  'B在{0}与{1}之间'.format('A','C');
 */
String.prototype.format = function () {
    // 数据长度为空，则直接返回
    if (arguments.length == 0) {
        return this;
    }

    // 使用正则表达式，循环替换占位符数据
    for (var result = this, i = 0; i < arguments.length; i++) {
        result = result.replace(new RegExp("\\{" + i + "\\}", "g"), arguments[i]);
    }
    return result;
};

function timingMaintain(){
	if(time>0){
		num++;
		if(time<=600){ //当维护时间小于10分钟时进入判断
			if(num==60){ //每分钟弹出一次维护公告
				ajaxAnnoument(); //弹出维护公告
				num=0;
			}

			if(num==300){//每五分钟弹出一次维护公告
				ajaxAnnoument(); //弹出维护公告
				num=0;
			}
		}
		time--;
	}
	setTimeout(timingMaintain,1000);

                var depositMoney = $("#depositMoney").val();

                var opreStandardCompany = $("#opreStandardCompany").val();

                //var numCompCompany = $("#numCompCompany").val();
                //存入金额为空
                if (null == depositMoney || depositMoney == "") {
                    $("#depositMoneymsgs").html("<font color=red>" + i18n.manDepositAndDraw.depositAmountTips + "</font>");
                    return;
                }
                if (!(/(^[+]?([0-9]*\.?[0-9]+|[0-9]+\.?[0-9]*)([eE][+]?[0-9]+)?$)/).test(depositMoney)) {
                    $("#depositMoneymsgs").html("<font color=red>" + i18n.manDepositAndDraw.wrongFormat + "</font>");

                    return;
                } else if (depositMoney <= 0) {
                    $("#depositMoneymsgs").html("<font color=red>" + i18n.manDepositAndDraw.depositAmountMinLimit + "</font>");
                    return;
                } else {
                    $("#depositMoneymsgs").html("<font color=green>" + i18n.manDepositAndDraw.correct + "</font>");
                }
                if (typeid == 1) {
                    //存入金额小于优惠标准
                    if (parseFloat(depositMoney) < parseFloat(preStandardCompany)) {
                        $("#depositPre").val("0");
                        $("#depositPre").attr('disabled', 'disabled');
                        $("#depositPreStatus").attr('disabled', 'disabled');
                        $("#compBet").val("0");
                        $("#compBet").attr('disabled', 'disabled');
                        $("#compBetCheckStatus").attr('disabled', 'disabled');
                    } else {
                        var res = parseFloat(depositMoney) * prePerCompany * 0.01;
                        if (parseFloat(res) >= parseFloat(preUpCompany)) {
                            res = preUpCompany;
                        }
                        $("#depositPre").val(parseFloat(res).toFixed(2));
                        $("#compBet").val("0");
                        $("#compBet").attr('disabled', 'disabled');
                        $("#compBetCheckStatus").attr('disabled', 'disabled');
                    }
                    //存入金额小于其他优惠标准
                    if (parseFloat(depositMoney) < parseFloat(opreStandardCompany)) {
                        $("#otherPre").val("0");
                        $("#otherPre").attr('disabled', 'disabled');
                        $("#otherPreStatus").attr('disabled', 'disabled');
                    }
                    if (parseFloat(depositMoney) < parseFloat(opreStandardCompany)) {
                        $("#otherPre").val("0");
                    } else {
                        var result = parseFloat(depositMoney) * oprePerCompany * 0.01;
                        if (result >= opreUpCompany) {
                            $("#otherPre").val(parseFloat(opreUpCompany).toFixed(2));
                        } else {
                            $("#otherPre").val(parseFloat(result).toFixed(2));
                        }
                    }

                }
